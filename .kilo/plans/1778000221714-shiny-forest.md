### Phase 2: Modern GitHub Updater (30 min)
**File**: `src/updater/github-release-update.ts` - REWRITE

Changes:
1. Use modern `autoUpdater` config with `provider: 'github'`
2. Support both public and private repos
3. Read config from `UPDATE_ENV` constants
4. Proper error handling and retry logic
5. Expose typed API for renderer via preload

```typescript
import { autoUpdater, type UpdateInfo } from 'electron-updater';
import { app } from 'electron';
import { UPDATE_ENV, UPDATE_IPC_CHANNELS } from '@/constants';

export interface UpdateConfig {
  provider: 'github' | 'private';
  channel: string;
  checkIntervalMs: number;
}

export class GitHubReleaseUpdater {
  private config: UpdateConfig;
  
  constructor(config: UpdateConfig) {
    this.config = config;
    this.configure();
  }
  
  private configure() {
    autoUpdater.autoDownload = true;
    autoUpdater.autoInstallOnAppQuit = true;
    autoUpdater.autoRunAppAfterInstall = true;
    
    // Modern GitHub provider configuration
    autoUpdater.setFeedURL({
      provider: 'github',
      owner: UPDATE_ENV.GITHUB_OWNER || 'your-username',
      repo: UPDATE_ENV.GITHUB_REPO || 'your-repo',
      private: !!UPDATE_ENV.GITHUB_TOKEN,
      token: UPDATE_ENV.GITHUB_TOKEN,
      channel: this.config.channel,
    });
  }
  
  async checkForUpdates(): Promise<UpdateInfo | null> {
    try {
      const result = await autoUpdater.checkForUpdates();
      return result.updateInfo;
    } catch (error) {
      console.error('[updater] Check failed:', error);
      return null;
    }
  }
  
  // ... rest of implementation
}
```

### Phase 3: Private Update System (45 min)
**File**: `src/updater/private-update.ts` - CREATE

Implement S3/CloudFront signed URL updater:
```typescript
import { autoUpdater } from 'electron-updater';
import { UPDATE_ENV } from '@/constants';

export class PrivateUpdateService {
  private authEndpoint: string;
  
  constructor() {
    this.authEndpoint = UPDATE_ENV.AUTH_ENDPOINT!;
  }
  
  async getSignedUpdateUrl(): Promise<string> {
    // Fetch temporary signed URL from auth endpoint
    const response = await fetch(
      `${this.authEndpoint}/v1/desktop/update-token?` +
      `platform=${process.platform}&arch=${process.arch}` +
      `&channel=${UPDATE_ENV.CHANNEL}&version=${app.getVersion()}`,
      {
        headers: {
          'Authorization': `Bearer ${UPDATE_ENV.AUTH_BEARER}`,
        },
      }
    );
    
    const data = await response.json();
    return data.baseUrl;
  }
  
  async configure(): Promise<void> {
    const feedUrl = await this.getSignedUpdateUrl();
    autoUpdater.setFeedURL({ url: feedUrl });
  }
}
```

### Phase 5: Preload API Integration (30 min)
**File**: `src/preload/updater.ts` - CREATE

```typescript
import { contextBridge, ipcRenderer } from 'electron';
import { UPDATE_IPC_CHANNELS } from '@/constants';

export interface UpdateInfo {
  version: string;
  releaseName?: string;
  releaseNotes?: string;
}

export interface UpdateStatus {
  checking: boolean;
  available: boolean;
  downloading: boolean;
  downloaded: boolean;
  error?: string;
  progress?: number;
}

contextBridge.exposeInMainWorld('electronAPI', {
  update: {
    checkForUpdates: () => ipcRenderer.invoke(UPDATE_IPC_CHANNELS.CHECK_FOR_UPDATES),
    onUpdateAvailable: (callback: (info: UpdateInfo) => void) => 
      ipcRenderer.on(UPDATE_IPC_CHANNELS.UPDATE_AVAILABLE, callback),
    onUpdateDownloaded: (callback: (info: UpdateInfo) => void) => 
      ipcRenderer.on(UPDATE_IPC_CHANNELS.UPDATE_DOWNLOADED, callback),
    onProgress: (callback: (progress: number) => void) => 
      ipcRenderer.on(UPDATE_IPC_CHANNELS.DOWNLOAD_PROGRESS, callback),
    installUpdate: () => ipcRenderer.send(UPDATE_IPC_CHANNELS.INSTALL_UPDATE),
  },
});
```

```typescript
export function UpdateNotification() {
  const [status, setStatus] = useState<UpdateStatus>({ checking: false, available: false, downloading: false, downloaded: false });
  
  useEffect(() => {
    window.electronAPI.update.onUpdateAvailable((info) => {
      setStatus({ ...status, available: true });
    });
  }, []);
  
  if (!status.available) return null;
  
  return (
    <div className="fixed bottom-4 right-4 bg-background border rounded-lg shadow-lg p-4">
      <h3 className="font-semibold">Update Available</h3>
      <p className="text-sm text-muted-foreground">A new version is ready to install</p>
      <Button onClick={() => window.electronAPI.update.installUpdate()}>
        Install Update
      </Button>
    </div>
  );
}
```
